62 uint8_t multiboot_free_memory() {
63  if (!mb_info || !mb_info->flags.mmap)
64  return 0;
65  uintptr_t mmap = mb_info->mmap_addr, // get the mmap start and end addresses
66  mmap_end = (uintptr_t) mb_info->mmap_addr + mb_info->mmap_length;
67  for (multiboot_memory_map_t* mmap_entry; // iterate over every mmap entry
68  mmap_entry = (multiboot_memory_map_t*) mmap, mmap < mmap_end;
69  // because offset 0 is not size but base_addr_low, we add size + 4 bytes
70  mmap += mmap_entry->size + sizeof(mmap_entry->size)) {
71  // We ignore the upper bytes here because we assume the machine does not
72  // have more than 4 GiB RAM. (Might be relevant if we wanted to use PAE.)
73  pmm_use((void*) mmap_entry->base_addr_low, mmap_entry->length_low,
74  mmap_entry->type == 1 ? PMM_UNUSED : PMM_RESERVED, "BIOS memory");
75  }
76  return 1;
77 }

79 void multiboot_copy_memory() {
80  size_t mb_info_length = sizeof(multiboot_info_t),
81  mmap_length = mb_info->mmap_length,
82  modules_length = mb_info->flags.mods ?
83  mb_info->mods_count * sizeof(multiboot_module_t) : 0;
84  // we allocate some space all at once to save (some) memory
85  void* dst = pmm_alloc(mb_info_length + mmap_length + modules_length, PMM_KERNEL);
86  // Now we copy some data we don't want to overwrite:
87  mb_info = memcpy(dst, mb_info, mb_info_length);
88  // the memory map (if we want to use it later)
89  mb_info->mmap_addr = (uint32_t) memcpy((void*) ((uintptr_t) dst + mb_info_length),
90  (void*) mb_info->mmap_addr, mmap_length);
91  if (mb_info->flags.mods) { // if there are modules, mark them as used
92  mb_info->mods_addr = memcpy((void*)
93  ((uintptr_t) dst + mb_info_length + mmap_length),
94  mb_info->mods_addr, modules_length);
95  for (int i = 0; i < mb_info->mods_count; i++) {
96  multiboot_module_t* module = mb_info->mods_addr + i;
97  size_t module_length = module->mod_end - module->mod_start + 1;
98  size_t string_length = strlen(module->string) + 1;
99  dst = pmm_alloc(module_length + string_length, PMM_KERNEL);
100  module->mod_start = memcpy(dst, module->mod_start, module_length);
101  module->string = memcpy((void*) ((uintptr_t) dst + module_length),
102  module->string, string_length);
103  }
104  }
