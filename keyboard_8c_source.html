33 // If the received scancode matches the given scancode or if the passed key
34 // string matches the given name, return a suitable keycode.
35 #define MAKE_KEYCODE(name, row, col, scancode...) { \
36  uint8_t compare_buf[SCANCODE_LEN] = { scancode }; \

102 static uint8_t keyboard_scancode_set(uint8_t scancode_set) {
103  if (scancode_set > 3)
104  return 0; // there is no scancode set 4, ...
105  ps2_write_device(port, SCANCODE_SET);
106  ps2_write_device(port, scancode_set);
107  if (scancode_set == 0) { // read scancode set rather than write
108  ps2_error_t err;
109  return ps2_read_device(port, &err);
110  }
111  ps2_config_t config = ps2_read_config();
112  config.bits.port1_transl = 0;
113  ps2_write_config(config);
114  return scancode_set;
115 }

117 static uint8_t keyboard_leds(keyboard_leds_t leds, int8_t value) {
118  static uint8_t led_state = 0;
119  if (value < 0)
120  led_state ^= leds; // toggle LEDs
121  else if (value > 0)
122  led_state |= leds; // enable LEDs
123  else
124  led_state &= ~leds; // disable LEDs
125  ps2_write_device(port, SET_LEDS);
126  ps2_write_device(port, led_state);
127  return led_state;
128 }

240 // called when a scancode has been received (makecode = key pressed)
241 static void keyboard_process_scancode(uint8_t is_makecode) {
242  event.keycode = keyboard_get_keycode_from_scancode(scancode_buf, scancode_len);
243  event.pressed = is_makecode;
244  // the pause key never sends a breakcode and we don't need lock breakcodes
245  // (in bochs they are not even sent, maybe because of the LED toggling)
246  if (event.keycode != KEY("PAUSE") && event.keycode != KEY("SCROLL") &&
247  event.keycode != KEY("NUM") && event.keycode != KEY("CAPS"))
248  keyboard_set_key_pressed(event.keycode, event.pressed);
249  event.ascii = keyboard_get_ascii_from_keycode(current_layout, event.keycode, event.flags);
250 
251  // handle modifier keys
252  event.flags.shift = keyboard_get_key_pressed(KEY("L SHFT")) ||
253  keyboard_get_key_pressed(KEY("R SHFT"));
254  event.flags.ctrl = keyboard_get_key_pressed(KEY("L CTRL")) ||
255  keyboard_get_key_pressed(KEY("R CTRL"));
256  event.flags.gui = keyboard_get_key_pressed(KEY("L GUI")) ||
257  keyboard_get_key_pressed(KEY("R GUI"));
258  event.flags.alt = keyboard_get_key_pressed(KEY("L ALT")) ||
259  keyboard_get_key_pressed(KEY("R ALT"));
260  if (event.pressed) {
261  if (event.keycode == KEY("SCROLL"))
262  keyboard_leds(SCROLL, event.flags.scroll_lock = !event.flags.scroll_lock);
263  if (event.keycode == KEY("NUM"))
264  keyboard_leds(NUM, event.flags.num_lock = !event.flags.num_lock);
265  if (event.keycode == KEY("CAPS"))
266  keyboard_leds(CAPS, event.flags.caps_lock = !event.flags.caps_lock);
267  }
268 
269  if (handler) handler(event); // call our event handler if there is one
270 
271  keyboard_state = START; // reset the state machine so
272  scancode_len = 0; // we can proceed with a fresh scancode
273 }

275 static uint8_t keyboard_process_special_scancode(uint8_t condition, uint8_t len,
276  uint8_t is_breakcode) {
277  if (condition) { // check whether a special scancode is being sent
278  keyboard_state = SPECIAL_RECEIVED;
279  if (scancode_len == len) // whether we fully received the scancode
280  keyboard_process_scancode(is_breakcode);
281  return 1;
282  }
283  return 0;
284 }

294 // called by the PS/2 driver with a scancode byte
295 void keyboard_handle_data(uint8_t data) {
296  if (keyboard_state == BREAK_RECEIVED) // ignore the BREAK_CODE byte
297  scancode_len--; // so we can easily convert breakcodes to keycodes
298  scancode_buf[scancode_len++] = data; // save scancode byte
299 
300  // handle special keys (pause and print screen)
301  if (keyboard_process_special_scancode(IS_PAUSE_MK(scancode_buf, scancode_len), 8, 1)) return;
302  if (keyboard_process_special_scancode(IS_PR_SC_MK(scancode_buf, scancode_len), 4, 1)) return;
303  if (keyboard_process_special_scancode(IS_PR_BR_MK(scancode_buf, scancode_len), 6, 0)) return;
304 
305  switch (keyboard_state) {
306  case START: // we just received the first scancode byte
307  if (data == BREAK_CODE) keyboard_state = BREAK_RECEIVED;
308  else if (data == EXTENDED_CODE) keyboard_state = EXTENDED_RECEIVED;
309  else keyboard_process_scancode(1); // received a one-byte scancode
310  break;
311  case BREAK_RECEIVED:
312  keyboard_process_scancode(0); // received a breakcode
313  break;
314  case EXTENDED_RECEIVED: // received an extended ...
315  if (data == BREAK_CODE)
316  keyboard_state = BREAK_RECEIVED; // ... breakcode
317  else keyboard_process_scancode(1); // ... scancode
318  break;
319  case SPECIAL_RECEIVED: // this will never be executed, prevent Wswitch
320  break;
321  }
322 }
