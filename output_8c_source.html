50 uint16_t io_putchar(uint8_t c) {
51  if (c == '\n')
52  cursor = cursor + IO_COLS - (cursor % IO_COLS);
53  else {
54  io_setchar(c, cursor);
55  io_setattr(attr, cursor);
56  cursor = (cursor + 1) % IO_CHARS;
57  }
58  return 1;
59 }

67 uint16_t log(char* tag, char* fmt, ...) {
