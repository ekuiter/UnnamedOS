51 void cpuid_name(cpuid_result_t* res) {
52  uint32_t buf[5];
53  *(uint32_t*) &buf[0] = res->eax;
54  *(uint32_t*) &buf[1] = res->ebx;
55  *(uint32_t*) &buf[2] = res->ecx;
56  *(uint32_t*) &buf[3] = res->edx;
57  buf[4] = '\0';
58  // the following code trims leading and trailing spaces of s
59  char* s = (char*) buf;
60  while (*s == ' ')
61  s++;
62  uint8_t spaces = 0;
63  for (; *s != '\0'; s++) {
64  if (*s == ' ')
65  spaces++;
66  else {
67  for (int i = 0; i < spaces; i++)
68  io_putchar(' ');
69  spaces = 0;
70  io_putchar(*s);
71  }
72  }
73 }
