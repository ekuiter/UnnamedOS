40 static uint8_t mouse_get_status(uint8_t* resolution, uint8_t* sample_rate) {
41  ps2_write_device(port, GET_STATUS);
42  ps2_error_t err;
43  uint8_t flags = ps2_read_device(port, &err);
44  // binary value->pixels per mm: 0->1, 1->2, 2->4, 3->8 (powers of 2)
45  *resolution = 1 << ps2_read_device(port, &err);
46  *sample_rate = ps2_read_device(port, &err); // decimal value (per second)
47  return flags;
48 }

50 static uint8_t mouse_resolution(uint8_t resolution) {
51  if (resolution == 0) { // read resolution
52  uint8_t sample_rate;
53  mouse_get_status(&resolution, &sample_rate);
54  return resolution;
55  }
56  else if (resolution == 1 || resolution == 2 || resolution == 4 ||
57  resolution == 8) { // write resolution
58  uint8_t resolution_value = 0;
59  while (resolution >>= 1) // pixels per mm->binary value:
60  resolution_value++; // 1->0, 2->1, 4->2, 8->3 (log 2)
61  ps2_write_device(port, SET_RESOLUTION);
62  ps2_write_device(port, resolution_value);
63  return resolution;
64  } else
65  return 0; // invalid resolution given
66 }

68 static uint8_t mouse_sample_rate(uint8_t sample_rate) {
69  if (sample_rate == 0) { // read sample rate
70  uint8_t resolution;
71  mouse_get_status(&resolution, &sample_rate);
72  return sample_rate;
73  }
74  else if (sample_rate == 10 || sample_rate == 20 || sample_rate == 40 ||
75  sample_rate == 60 || sample_rate == 80 || sample_rate == 100 ||
76  sample_rate == 200) { // write sample rate
77  ps2_write_device(port, SET_SAMPLE_RATE);
78  ps2_write_device(port, sample_rate);
79  return sample_rate;
80  } else
81  return 0; // invalid sample rate given
82 }

98 static void mouse_process_packet(mouse_packet_t* packet) {
99  packet_len = 0; // ready for receiving a new packet
100  if (packet->x_overflow || packet->y_overflow)
101  return; // ignore overflowing packets
102  int16_t dx = packet->x_movement, // assume normal positive integer, so fill
103  dy = packet->y_movement; // high byte up with 0: 0x00xm and 0x00ym
104  if (packet->x_sign) // x_movement (xm) contains two's complement, we need to
105  dx |= 0xFF00; // extend dx to a 2-byte negative number of form 0xFFxm
106  if (packet->y_sign) // similar for y-axis
107  dy |= 0xFF00;
108  dy = -dy; // dy is positive when moving up, so we flip its sign
109  if (ABS(dx) > dx_max || ABS(dy) > dy_max)
110  return; // if a delta value is suspiciously high, ignore packet
111  event.x = SUM_TOO_LOW(x, dx, 0) ? 0 : (SUM_TOO_HIGH(x, dx, x_max) ? x_max : event.x + dx),
112  event.y = SUM_TOO_LOW(y, dy, 0) ? 0 : (SUM_TOO_HIGH(y, dy, y_max) ? y_max : event.y + dy);
113  event.screen_x = map(event.x, 0, x_max, 0, screen_width);
114  event.screen_y = map(event.y, 0, y_max, 0, screen_height);
115  event.left = packet->left;
116  event.right = packet->right;
117  event.middle = packet->middle;
118  if (handler) handler(event); // call our event handler if there is one
119 }

121 void mouse_handle_data(uint8_t data) {
122  packet_buf[packet_len++] = data; // save packet byte
123 
124  if (packet_len == 1 && !(data & FLAG_ALWAYS_ONE)) // if the "always one"
125  packet_len = 0; // flag is not set, something went wrong, so start over
126 
127  if (packet_len == PACKET_LEN) // if all bytes received, process packet
128  mouse_process_packet((mouse_packet_t*) packet_buf);
129 }
