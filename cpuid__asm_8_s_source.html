1 .global cpuid_check
2 cpuid_check:
3 // We need to check whether CPUID is available on this machine. To do that, we try to invert the ID flag
4 // in EFLAGS. If we succeed, CPUID is available, otherwise it's not.
5  pushf // push EFLAGS to stack
6  pop %ecx // pop EFLAGS off to ECX
7  mov %ecx, %eax // copy EFLAGS to eax
8  xor $0x200000, %eax // invert ID flag in EFLAGS
9  push %eax // push altered EFLAGS to stack
10  popf // pop altered EFLAGS to EFLAGS, we now need to check whether the change succeeded
11  pushf // push the (possibly) altered EFLAGS to stack
12  pop %eax // pop EFLAGS off to EAX, we now have: unaltered EFLAGS in ECX, (possibly) altered EFLAGS in EAX
13  xor %ecx, %eax // compare both EFLAGS
14  ret // if not equal, a nonzero value is returned in EAX and CPUID is available

16 .global cpuid_call
17 cpuid_call:
18 // We want to call CPUID. The function to call is given as the 1st argument, the 2nd argument is a pointer where to store the result.
19 // Because this function is called from C, we need to create and destroy a proper stack frame. We also need to preserve ESI because
20 // we change it inside the function.
21  push %ebp // create stack frame (store old frame pointer,
22  mov %esp, %ebp // let new frame pointer point to TOS)
23  push %esi // save ESI so we can use it in the meanwhile
24 
25  mov 8(%ebp), %eax // Transfer 1st argument (4 byte) to EAX. (%ebp) is old %ebp, 4(%ebp) is return address, 8(%ebp) is 1st argument.
26  cpuid // perform CPUID call with the function specified in EAX, results are in EAX, EBX, ECX, EDX
27  mov 12(%ebp), %esi // save pointer (2nd argument) to ESI
28  mov %eax, (%esi) // store EAX to pointer,
29  mov %ebx, 4(%esi) // EBX to pointer + 4, ...
30  mov %ecx, 8(%esi)
31  mov %edx, 12(%esi)
32  mov %esi, %eax // return the result pointer
33 
34  pop %esi // restore old ESI
35  mov %ebp, %esp // destroy stack frame (let TOS point to current frame pointer,
36  pop %ebp // restore old frame pointer)
37  ret
